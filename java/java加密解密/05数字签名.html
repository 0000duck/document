签名多以十六进制数据显示
以公钥解密 取得摘要 和摘要对比验证

example

final String KEY_ALGORITHM="RSA";	//算法名主要用于产生密钥对		
final int KEY_SIZE=1024;				//密钥长度
		
final String  SIGNATURE="MD5withRSA";	//签名算法名
		
KeyPairGenerator keyPairGen=KeyPairGenerator.getInstance(KEY_ALGORITHM);	
keyPairGen.initialize(KEY_SIZE);						//初始化密钥工厂
KeyPair keyPair = keyPairGen.generateKeyPair();			//生成密钥对	
		
String str="我是被签名的数据";
System.out.println("数据:"+str);
		
//签名
Signature signature=Signature.getInstance(SIGNATURE);此类需要导入java 自带的包 而非BC
signature.initSign(keyPair.getPrivate());
signature.update(str.getBytes());
byte[] data=signature.sign();
System.out.println("签名:"+new String(Hex.encode(data)));
		
//验证
signature=Signature.getInstance(SIGNATURE);
signature.initVerify(keyPair.getPublic());
signature.update(str.getBytes());
if(signature.verify(data))
{
	System.out.println("验证:pass");
}
else
{
	System.out.println("验证:no pass");
}
		
		
/*---------------我是无奈的分割线--------------*/	
		
str="我是被修改过的签名的数据";
System.out.println("数据:"+str);
signature.initVerify(keyPair.getPublic());
signature.update(str.getBytes());
if(signature.verify(data))
{
	System.out.println("验证:pass");
}
else
{
	System.out.println("验证:no pass");
}
}
