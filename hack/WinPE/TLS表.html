TLS表記錄了TLS信息 其為一個 IMAGE_TLS_DIRECTORY 結構 
typedef struct _IMAGE_TLS_DIRECTORY32 {
    DWORD   StartAddressOfRawData;		//TLS數據塊開始地址
    DWORD   EndAddressOfRawData;		//TLS數據塊結束地址(不包含用0填充的數據)
    DWORD   AddressOfIndex;             // PDWORD					索引地址數組
    DWORD   AddressOfCallBacks;         // PIMAGE_TLS_CALLBACK *	TLS回調地址數組	以0結尾
    DWORD   SizeOfZeroFill;				//用0填充大小
    DWORD   Characteristics;			//保留未使用
} IMAGE_TLS_DIRECTORY32;
typedef IMAGE_TLS_DIRECTORY32 * PIMAGE_TLS_DIRECTORY32;
