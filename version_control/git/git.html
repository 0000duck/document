git

git由 linus(linux 之父) 創作

proxy 

//設置代理
git config --global http.proxy 'socks5://127.0.0.1:1080'
git config --global https.proxy 'socks5://127.0.0.1:1080'

git config --global https.proxy http://127.0.0.1:1080
git config --global https.proxy https://127.0.0.1:1080

//取消代理
git config --global unset http.proxy
git config --global --unset https.proxy

git 初始化 

//顯示 git 版本
git --version



//配置 git 提交 用戶名 郵件地址
git config --global user.name "zuiwuchang"
git config --global user.email "zuiwuchang@gmail.com"
	

//在git命令輸出中 開啟 顏色 顯示
git config --global color.ui true

//讀取當前生效的 git 配置值
git config section.key
	git config user.name
//設置當前生效的 git 配置值
git config section.key value
	git config user.name zuiwuchang



//在 當前目錄/指定目錄 創建 git 版本庫
git init [--bare] [path]
	--bare 創建裸版本庫
	path 版本庫 位置 默認 ./

//將 文件 增加到 暫存區
git add [file] [-u] [-i]
	-u 將本地 有改變的(包括 刪除 修改 新增) 操作都加入到 暫存區
	-i 以交互方式 添加

//將 文件 從版本庫中 刪除
git rm [file]

//將 文件 從版本庫中 改名
git mv [source] [dist]

//提交版本
git commit [--amend] [--allow-empty] -m "commit text"
	-m 設置 提交 備註
	--amend 對剛剛的提交 修改備註
	--allow-empty 允許 不設置 用戶 郵件 提交 默認不允許
	

//查看 提交 記錄
git log [--stat] [--pretty=oneline] [-l HEAD]
	--stat 顯示變更的 內容
	--pretty=oneline 以精簡的方式 顯示 記錄
	--pretty=raw ... 詳細 ...
	-l 只顯示 指定分支

//獲取一份 完整的 git 克隆
git clone source [dist]



//在工作區 文件中 搜索 指定 內容
git grep "搜索內容"

//顯示 .git 所在目錄
git rev-parse --git-dir

//顯示 工作區目錄
git rev-parse --show-toplevel

//顯示 當前目錄 相對 工作區目錄
git rev-parse --show-prefix

//顯示 當前目錄 返回到 工作區目錄 的深度
git rev-parse --show-cdup




//查看 git 對象
git cat-file [-t] [-s] [-p] obj 
	obj	要查看的對象
	-t	對象型別
	-s	對象大小
	-p	對象數據

注意 

git config --global
記錄在 ~/.gitconfig 文件中

git config
記錄在 當前版本庫的 .git/config 文件中

git config --system
記錄在 /etc/gitconfig 文件中



git config 命令 其實 可以操作 任何 ini 文件
GIT_CONFIG=test.ini git config a.b "test ini"
GIT_CONFIG=test.ini git config a.b

暫存區 

//將 工作區的 文件 加入到 暫存區
git add path



//比較 工作區 和 暫存區  的 差異
git diff [要比較多 分支]
	[要比較多 分支] 不寫和 暫存區比較
	HEAD	當前分支
	


//查看 工作區 狀態
git status [-s] [-b] 
	-s 以精簡的方式 顯示
	-b 顯示當前分支 名



//將 暫存區的 內容 以 HEAD指示的 版本 替換
git reset HEAD

//將 暫存區 的 指定文件 刪除
git rm --cached <file>

//用 暫存區 全部文件/指定文件 替換工作區
git checkout  ./-- <file>

//用 指定分支 全部文件/指定文件 替換 暫存區 
git checkout HEAD [<file>]



//將 工作區 和暫存區相比  沒有 改變的 全刪除
git clean -fd

git reset 重置 

git reset 用與將 操作 版本游標(HEAD)



HEAD 指向最近的 一次 提交
^ 其父提交

//將工作區 暫存區 恢復到指定 游標
git reset --hard HEAD
git reset --hard {master}@{n}	//引用 git reflog 中的id值


//將 暫存區 恢復到指定 游標
git reset HEAD
git reset

//將 filename 移除 暫存區
git reset HEAD filename


//僅改變 版本庫 游標位置
git reset --soft



//顯示 版本庫操作記錄
git reflog [show master]
	show master 只顯示 master分支 記錄

git checkout 檢出 

//將 HEAD 指向 指定分支
git checkout branch

//匯總 暫存區 工作區 與 HEAD 的差別
git checkout
git checkout HEAD

//用暫存區的 filename 替換 工作區的 filename
git checkout -- filename

//維持 HEAD 不變 用 branch 的 filename 替換 暫存區 工作區 的 filename
git checkout branch -- filename


//用 暫存區的 所有文件 替換 工作區
git checkout .

git stash 進度 

//測試 要刪除的 本地 多餘 文件 目錄
git clean -nd

//刪除 本地 多餘的 文件 目錄
git clean -fd



//保存 進度(暫存區 工作區)
git stash

//顯示 保存的進度 列表
git stash list

//恢復進度(默認恢復最新進度) 並刪除此進度
git stash pop [--index] [<stash>]
	--index 恢復工作區的同時 嘗試恢復 暫存區
	<stash> git stash list中的值

//同 pop 不過不會刪除 進度罷
git stash apply

//刪除 進度(默認刪除 最新進度)
git stash drop [<stash>]

//刪除 所有 進度
git stash clear

//基於進度 創建分支
git stash branch [<branchname>] [<stash>]
