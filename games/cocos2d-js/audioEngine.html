audioEngine

audioEngine 

Class
cc.audioEngine

Defined in: CCAudio.js

用處 

audioEngine 全部由 static 方法組成 提供了一個 
全局的 音頻播放 單件

audioEngine 同時 只播放一個 音樂(遊戲背景) 和混合多個 特效(音效)

方法 

//結束 音樂 和 特效
<static> cc.audioEngine.end()



//獲取 特效/音樂 音量 [0.0,1.0]
<static> {Number} cc.audioEngine.getEffectsVolume()
<static> {Number} cc.audioEngine.getMusicVolume()

//設置 特效/音樂 音量 [0.0,1.0]
<static> cc.audioEngine.setEffectsVolume(volume)
<static> cc.audioEngine.setMusicVolume(volume)



//播放 特效(返回 audioID) 音樂
<static> {Number|null} cc.audioEngine.playEffect(url, loop = false)
<static> cc.audioEngine.playMusic(url, loop = false)

//暫停 所有特效/指定特效 音樂
<static> cc.audioEngine.pauseAllEffects()
<static> cc.audioEngine.pauseEffect(audioID)
<static> cc.audioEngine.pauseMusic()

//恢復 所有特效/指定特效 音樂
<static> cc.audioEngine.resumeAllEffects()
<static> cc.audioEngine.resumeEffect(audioID)
<static> cc.audioEngine.resumeMusic()

//停止 播放 所有特效/指定特效 音樂(releaseData=true 釋放音樂)
<static> cc.audioEngine.stopAllEffects()
<static> cc.audioEngine.stopEffect(audioID)
<static> cc.audioEngine.stopMusic(releaseData = false)



//倒帶 音樂 到 開始
<static> cc.audioEngine.rewindMusic()

//音樂 處於 播放 狀態
<static> {Boolean} cc.audioEngine.isMusicPlaying()

//釋放 特效
<static> cc.audioEngine.unloadEffect(url)

//返回 背景 音樂是否 可播放
<static> {boolean} cc.audioEngine.willPlayMusic()
