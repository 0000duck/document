	為窗口插入對話框資源 必須為 IDD_FORMVIEW 資源 並添加類 必須從CFormView繼承

	CSplitterWnd m_splitter;	//定義一個分割器
	CFrameWnd* m_frame;			//一個動態創建的視圖

3    
//注册窗口类
    CString strMyClass = AfxRegisterWndClass(CS_VREDRAW | CS_HREDRAW,  
                   ::LoadCursor(NULL, IDC_ARROW),    (HBRUSH) ::GetStockObject(WHITE_BRUSH),   
                  ::LoadIcon(NULL, IDI_APPLICATION));
    // 創建CFrameWnd
    m_frame = new CFrameWnd;
    m_frame->Create(strMyClass,L"", WS_CHILD,   CRect(0,0,300,300), this);
   

    //創建分隔器
    m_splitter.CreateStatic(m_frame,1, 2);		//在Frame里切分视图窗口为1*2，就是一行两列
	//將要顯示的窗體添加到分割器
	m_splitter.CreateView(0,0, RUNTIME_CLASS(CForm1),   CSize(100,100), NULL);//第一行一列
    m_splitter.CreateView(0,1, RUNTIME_CLASS(CFrom2), CSize(100,100), NULL);//第一行二列

4 顯示窗口
CRect cRect;
	GetWindowRect(cRect);
	ScreenToClient(cRect);
	m_frame->MoveWindow(cRect);
	m_frame->ShowWindow(SW_SHOW);
