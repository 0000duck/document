使用 需要將 Markup.h Markup.cpp 加入項目 官网:http://www.firstobject.com/

CMarkup xml;
	//載入xml文件
	if(xml.Load(L".setting.cnf") && xml.FindElem(L"root"))
	{
		//查找子節點
		while(xml.FindChildElem(L"war"))
		{
			if(xml.GetChildAttrib(L"name")==name)
			{
				//刪除子節點
				xml.RemoveChildElem();
				break;
			}
		}
        //重置子節點查找位置
        xml.ResetChildPos();
		while(xml.FindChildElem(L"war3"))
		{
			if(xml.GetChildAttrib(L"name")==name)
			{
				xml.RemoveChildElem();
				break;
			}
		}
