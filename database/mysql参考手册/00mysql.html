mysql 

//導出 數據庫/表
mysqldump -u root -p --default-character-set=utf8 數據庫名 > 備份文件名
mysqldump -u root -p --default-character-set=utf8 數據庫名 表名 > 備份文件

//導入
source 備份文件名
mysql -u username -p -D dbname < filename.sql
mysqldump -u username -p dbname < filename.sql

//連接
mysql -h localhost -P 3306 -u uname -p --default-character-set=utf8 "--defaults-file=C:\ProgramData\MySQL\MySQL Server 5.6\my.ini"

//允許 某個用戶 遠程登錄
GRANT ALL PRIVILEGES ON *.* TO 'name'@'%' IDENTIFIED BY 'password' WITH GRANT OPTION;
FLUSH PRIVILEGES;

//設置 mysql 使用utf8 字符集
[client]
default-character-set=utf8
[mysqld]
character_set_server   = utf8
init_connect='SET NAMES utf8'		//(可選)

//查看使用 字符集
show variables like 'character%';
