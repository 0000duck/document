example 

#include <iostream>
#include<cstdio>

#include<Poco/UTF8String.h>

int main()
{
   std::string utf8    =   "This is utf8 test";

   puts("************   utf8    *************");
   std::cout<<utf8<<"\n";

   std::cout<<Poco::UTF8::icompare(utf8,"This is utf8 test")<<"\n";

   puts("************   toLower    *************");
   std::string str  = Poco::UTF8::toLower(utf8);
   std::cout<<str<<"\n"<<utf8<<"\n";

   puts("************   toLowerInPlace    *************");
   str  = Poco::UTF8::toLowerInPlace(utf8);
   std::cout<<str<<"\n"<<utf8<<"\n";

   puts("************   toUpper    *************");
   str  = Poco::UTF8::toUpper(utf8);
   std::cout<<str<<"\n"<<utf8<<"\n";

   puts("************   toUpperInPlace    *************");
   str  = Poco::UTF8::toUpperInPlace(utf8);
   std::cout<<str<<"\n"<<utf8<<"\n";

   return 0;
}
