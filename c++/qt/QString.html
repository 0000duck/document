QTextStream 

#include <QTextStream>

QTextStream 是一個 文本 輸入 輸出流 可以將 文本 在指定設備上 以流的方式 操作

example 

#include <QDebug>
#include <QString>
#include <QTextStream>

int main()
{
    //定義 輸出設備
    QString device;
    //定義流
    QTextStream stream(&device,QIODevice::ReadWrite);
    //輸出
    stream<<"line "<<1;
    stream<<" kata beckinsale is so sexy\n";

    qDebug()<<device;//"line 1 kata beckinsale is so sexy\n"

    QString cmd = stream.readAll();
    qDebug()<<cmd;//"line 1 kata beckinsale is so sexy\n"

    return 0;
}
