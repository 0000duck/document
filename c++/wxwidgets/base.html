win32 環境配置 wxwidgets

win32 環境配置 

1	安裝MinGw
2	安裝codeblocks
3	下載最新 wxwidgets 源碼 並解壓
4	cd wxwidgets/build/msw 編譯 wxwidgets
	//編譯 動態庫
	mingw32-make -f makefile.gcc MONOLITHIC=0 SHARED=1 UNICODE=1 BUILD=release
	mingw32-make -f makefile.gcc MONOLITHIC=0 SHARED=1 UNICODE=1 BUILD=debug
	//編譯 靜態庫
	mingw32-make -f makefile.gcc MONOLITHIC=0 SHARED=0 UNICODE=1 BUILD=release
	mingw32-make -f makefile.gcc MONOLITHIC=0 SHARED=0 UNICODE=1 BUILD=debug

	UNICODE(1/0 是否使用 unicode)
	MONOLITHIC(1/0 是否將庫編譯到一個文件中)
	SHARED (1/0 編譯動態庫/靜態庫)
	USE_ODBC (1/0 是否使用 odbc)

5	打開codeblock 增加全局 變量 wx

注意 

需要編譯 debug 版本庫 才能編譯debug版本項目

wxwidgets 

wxwidgets 運行 流程
1	依據不同平臺 創建不同入口函數 
	並實例化一個 wxApp(代表一個應用程序實例) 對象

2	調用 wxApp::OnInit()  在次函數中創建一個 wxFrame(第一個窗口) 的實例 並且顯示

3	wxApp::OnInit() 返回 true
	返回 false 程序自動 退出

4	wxFrame 的構造函數 被調用 在此爲wxFrame創建 菜單狀態欄等
	程序員可以在此 爲wxFrame增加 自定義的 佈局 ui 綁定事件等 

5	進入事件循環 處理

/***************************************************************
 * Name:      wxTestMain.cpp
 * Purpose:   Code for Application Frame
 * Author:    illusive man (zuiwuchang@gmail.com)
 * Created:   2014-08-25
 * Copyright: illusive man (http://zuiwuchang.blogspot.com/)
 * License:
 **************************************************************/

#ifdef WX_PRECOMP
#include "wx_pch.h"
#endif

#ifdef __BORLANDC__
#pragma hdrstop
#endif //__BORLANDC__

#include "wxTestMain.h"

//helper functions
enum wxbuildinfoformat {
    short_f, long_f };

wxString wxbuildinfo(wxbuildinfoformat format)
{
    wxString wxbuild(wxVERSION_STRING);

    if (format == long_f )
    {
#if defined(__WXMSW__)
        wxbuild << _T("-Windows");
#elif defined(__WXMAC__)
        wxbuild << _T("-Mac");
#elif defined(__UNIX__)
        wxbuild << _T("-Linux");
#endif

#if wxUSE_UNICODE
        wxbuild << _T("-Unicode build");
#else
        wxbuild << _T("-ANSI build");
#endif // wxUSE_UNICODE
    }

    return wxbuild;
}

//靜態綁定 事件
BEGIN_EVENT_TABLE(wxTestFrame, wxFrame)
    EVT_CLOSE(wxTestFrame::OnClose)
    EVT_MENU(idMenuQuit, wxTestFrame::OnQuit)
    EVT_MENU(idMenuAbout, wxTestFrame::OnAbout)
END_EVENT_TABLE()

wxTestFrame::wxTestFrame(wxFrame *frame, const wxString& title)
    : wxFrame(frame, -1, title)
{
//創建狀態欄
#if wxUSE_MENUS
    // create a menu bar
    wxMenuBar* mbar = new wxMenuBar();
    wxMenu* fileMenu = new wxMenu(_T(""));
    fileMenu->Append(idMenuQuit, _("&Quit\tAlt-F4"), _("Quit the application"));
    mbar->Append(fileMenu, _("&File"));

    wxMenu* helpMenu = new wxMenu(_T(""));
    helpMenu->Append(idMenuAbout, _("&About\tF1"), _("Show info about this application"));
    mbar->Append(helpMenu, _("&Help"));

    SetMenuBar(mbar);
#endif // wxUSE_MENUS

//創建菜單
#if wxUSE_STATUSBAR
    // create a status bar with some information about the used wxWidgets version
    CreateStatusBar(2);
    SetStatusText(_("Hello Code::Blocks user!"),0);
    SetStatusText(wxbuildinfo(short_f), 1);
#endif // wxUSE_STATUSBAR

}


wxTestFrame::~wxTestFrame()
{
}

void wxTestFrame::OnClose(wxCloseEvent &event)
{
    Destroy();
}

void wxTestFrame::OnQuit(wxCommandEvent &event)
{
    Destroy();
}

void wxTestFrame::OnAbout(wxCommandEvent &event)
{
    wxString msg = wxbuildinfo(long_f);
    wxMessageBox(msg, _("Welcome to..."));
}
