wxFileDialog 

wxFileDialog 是一個文件選擇對話框
提供用戶 選擇 1個/多個 文件
或 保存/打開 文件 的功能

常用接口 

//構造函數
wxFileDialog	(	wxWindow * 	parent,	//父窗口
	const wxString & 	message = wxFileSelectorPromptStr,	//標題
	const wxString & 	defaultDir = wxEmptyString,	//默認文件路徑
	const wxString & 	defaultFile = wxEmptyString,	//默認文件名
	const wxString & 	wildcard = wxFileSelectorDefaultWildcardStr,	//文件過濾器
	long 	style = wxFD_DEFAULT_STYLE,
	const wxPoint & 	pos = wxDefaultPosition,
	const wxSize & 	size = wxDefaultSize,
	const wxString & 	name = wxFileDialogNameStr 
)	

//顯示 對話框 用戶確認操作 返回wxID_OK 
int ShowModal	(		)	

//返回 選擇的 文件名
wxString 	GetFilename () const

//返回 選擇的 文件名 全路徑
virtual wxString 	GetPath () const

wxDirDialog

wxDirDialog 是一個 文件夾 選擇窗口


		

		
example

void darkDialogBack::OnButtonDirectory(wxCommandEvent& event)
{
    wxDirDialog dlg(this);
    if(dlg.ShowModal() == wxID_OK)
    {
        wxString path = dlg.GetPath();
        wxMessageBox(path);
    }

}
