package main

import (
	"fmt"
	"log"
	"net"
)

func main() {
	s, e := net.Listen("tcp", ":1102")
	if e != nil {
		log.Fatalln(e)
		return
	}
	for {
		c, e := s.Accept()
		if e != nil {
			log.Println(e)
			continue
		}
		go read(c)

	}
}
func read(c net.Conn) {
	fmt.Printf("%v in\n", c.RemoteAddr())
	buf := make([]byte, 1024)

	for {
		n, e := c.Read(buf)
		if e != nil {
			fmt.Printf("%v out\n", c.RemoteAddr())
			return
		}

		fmt.Println("%v", string(buf[:n]))
	}
}

package main

import (
	"fmt"
	"log"
	"net"
)

func main() {
	c, e := net.Dial("tcp", "127.0.0.1:1102")
	if e != nil {
		log.Fatalln(e)
	}

	var cmd string
	for {
		fmt.Print(">")
		fmt.Scan(&cmd)
		c.Write([]byte(cmd))
	}
}
