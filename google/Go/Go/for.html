if 

//if用法 同c 不过 可以在if前 加一个 便捷语句 其作用域在if内
	i := 0
	if v := 100; i == 0 {
		fmt.Println("0", v)
	} else if i < 0 {
		fmt.Println("min", v)
	} else {
		fmt.Println("max", v)
	}

注意 

else 必须跟在 }之后 若换行 一些编译器可能无法正确识别

switch 

//swith 基本同 c 不过和if一样 可以多个 便捷语句
import (
	"fmt"
	"runtime"
)

func main() {
	fmt.Print("Go runs on ")
	switch os := runtime.GOOS; os {
	case "darwin":
		fmt.Println("OS X.")
	case "linux":
		fmt.Println("Linux.")
	default:
		// freebsd, openbsd,
		// plan9, windows...
		fmt.Printf("%s.", os)
	}
}

switch 

//switch 不传条件 相当于 switch true (可以使用更清晰的方式 替换掉 if-else 链)
    import (
	"fmt"
	"time"
)

func main() {
	t := time.Now()
	switch {
	case t.Hour() < 12:
		fmt.Println("Good morning!")
	case t.Hour() < 17:
		fmt.Println("Good afternoon.")
	default:
		fmt.Println("Good evening.")
	}
}

注意
