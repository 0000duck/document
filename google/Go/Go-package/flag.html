flag 

flag 包 提供了 對命令行 參數 進行解析 的 功能



flag 支持 下面 三種 參數 格式
-flag
-flag=x
-flag x  // non-boolean flags only(不能是 bool)
	bool 取值
	1, 0, t, f, T, F, true, false, TRUE, FALSE, True, False

example 

package main

import (
	"flag"
	"fmt"
)

func main() {
	//定義要解析的 參數
	ip := flag.String("host", "127.0.0.1", "remote host")
	/*
		var ip string
		flag.StringVar(&ip,"host", "127.0.0.1", "remote host")
	*/
	h := flag.Bool("h", false, "for help")

	//解析參數
	flag.Parse()

	if *h {
		flag.PrintDefaults()
		return
	}

	fmt.Println(*h, *ip)
}
