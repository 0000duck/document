//返回文件是否符合指定 匹配模式
func Match(pattern, name string) (matched bool, err error)
	pattern:
		{ term }
	term:
		'*'         matches any sequence of non-/ characters
		'?'         matches any single non-/ character
		'[' [ '^' ] { character-range } ']'
		            character class (must be non-empty)
		c           matches character c (c != '*', '?', '\\', '[')
		'\\' c      matches character c

	character-range:
		c           matches character c (c != '\\', '-', ']')
		'\\' c      matches character c
		lo '-' hi   matches character c for lo <= c <= hi
	


//分割 文件夾 最後一個文件/文件夾 (如果存在最後一個的話 比如 /)
func Split(path string) (dir, file string)

package main

import (
	"fmt"
	"os"
	"os/exec"
	"path"
	"path/filepath"
)

func main() {

	p, _ := exec.LookPath(os.Args[0])
	p, _ = filepath.Abs(p)

	fmt.Println(p)

	fmt.Println("Base test")
	fmt.Println(path.Base(p))

	fmt.Println("Clean test")
	p1 := p + "/../../."
	fmt.Println(path.Clean(p1))

	fmt.Println("Dir test")
	fmt.Println(path.Dir(p1))
	p1 += "/123"
	fmt.Println(path.Dir(p1))

	fmt.Println("Ext test")
	fmt.Println(path.Ext(p1))
	p1 += ".exe"
	fmt.Println(path.Ext(p1))

	fmt.Println("IsAbs test")
	fmt.Println(path.IsAbs(p1))

	fmt.Println("Join test")
	p1 = path.Join(p, "123", "abc")
	fmt.Println(p1)

	fmt.Println("Match test")
	pattern := "g*"
	fmt.Println(path.Match(pattern, "golang"))

	fmt.Println("Splite test")
	d, f := path.Split(p)
	fmt.Println(d, f)
	p = "/"
	d, f = path.Split(p)
	fmt.Println(d, f)
}

注意 

path 不會管 文件是否存在 其只是更具 語義進行計算

package main

import (
	"fmt"
	"log"
	"os"
	"path/filepath"
)

func main() {
	e := filepath.Walk("d:/",
		func(path string, f os.FileInfo, e error) error {
			if f == nil {
				return e
			}
			if f.IsDir() {
				return nil
			}

			if f.Name() == "kate.jpg" {
				e := fmt.Errorf("find kate %v", path)
				fmt.Println(e)
				//return filepath.SkipDir
				//跳過 目錄

				return e
			}
			return nil
		})
	if e != nil {
		log.Fatalln(e)
	}

}

package main

import (
	"log"
	"os"
)

func main() {
	var dir = "d:/"
	f, e := os.Open(dir)
	if e != nil {
		log.Fatalln(e)
	}
	defer f.Close()
	info, e := f.Stat()
	if e != nil {
		log.Fatalln(e)
	}
	if !info.IsDir() {
		log.Fatalf("%v is not directory", dir)
	}

	if infos, e := f.Readdir(0); e != nil {
		log.Fatalln(e)
	} else {
		for _, info := range infos {
			if doWalk(dir, info) {
				break
			}
		}
	}
}
func doWalk(dir string, info os.FileInfo) bool {
	path := dir + info.Name()
	if info.IsDir() {
		f, e := os.Open(path)
		if e != nil {
			log.Println(e)
		}
		defer f.Close()
		if infos, e := f.Readdir(0); e != nil {
			log.Println(e)
		} else {
			for _, info := range infos {
				if doWalk(path+"/", info) {
					break
				}
			}
		}

	} else {
		if info.Name() == "kate.jpg" {
			log.Println("find kate ", path)
			return true
		}
	}
	return false
}
