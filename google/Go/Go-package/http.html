)

	Count     int
}

//POST 文件 example
func PostFile(info *Info, file string) error {
	//打開文件
	f, e := os.Open(file)
	if e != nil {
		return e
	}
	defer f.Close()
	//創建body
	body := &bytes.Buffer{}

	//創建 part
	writer := multipart.NewWriter(body)
	//Content-Disposition: form-data; name="file"; filename="1.jpg"
	part, err := writer.CreateFormFile("file", file)
	if err != nil {
		return err
	}
	//複製 byte 數據
	_, err = io.Copy(part, f)

	err = writer.Close()
	if err != nil {
		return err
	}

	//創建 request 請求
	url := fmt.Sprintf("http://%v/%v", info.Url, info.Fid)
	request, e := http.NewRequest("POST", url, body)
	if e != nil {
		return e
	}
	//設置 Content-Type
	request.Header.Set("Content-Type", writer.FormDataContentType())

	//發送POST請求
	client := &http.Client{}
	r, err := client.Do(request)
	if err != nil {
		return nil
	}

	//處理 回傳結果
	b, e := ioutil.ReadAll(r.Body)
	if e != nil {
		return e
	}
	fmt.Println(string(b))

	return nil
}

//Get Example
func GetVolume(info *Info) error {
	r, e := http.Get("http://192.168.1.254:9333/dir/assign")
	if e != nil {
		return e
	}
	b, e := ioutil.ReadAll(r.Body)
	if e != nil {
		return e
	}
	e = json.Unmarshal(b, &info)
	if e != nil {
		return e
	}
	return nil
}
