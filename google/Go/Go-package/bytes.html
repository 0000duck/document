bytes Buffer

bytes 

bytes 實現了 對 []byte 的 相關 操作

Buffer 

Buffer 是一個 []byte 輸入輸出 流

example 

package main

import (
	"bytes"
	"fmt"
	"os"
)

func main() {
	var buf bytes.Buffer
	buf.WriteString("cerberus it's an idea") //21 bytes
	fmt.Println(buf.Bytes())                 //返回 []byte
	fmt.Println(buf.Len())                   //21
	buf.WriteTo(os.Stdout)
	fmt.Println()
	fmt.Println(buf.Len()) //0
}
