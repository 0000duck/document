template 

Revel 的 template 使用 go 標準庫中 的 html/template
同時 加上了 幾個 Revel 定義的 模板 函數

if 

//.err 被定義 並且不爲 空
{{if .err}}
	error
{{else}}
	success
{{end}}
	

//.err 未定義 將引發錯誤
//.err 不爲空
{{if eq .err ""}}
{{else}}
	不爲空 執行 
{{end}}

for 

//遍歷 slice map 將 val 設置爲 當前環境 . 
{{range .nodes}}
	val = {{.}}
{{end}}

//同上 然 將 key val 設置到 指定 變量
{{range $key,$val := .nodes}}
	key = {{$key}}
	val = {{$val}}
{{end}}

	1

	
{{end}}
