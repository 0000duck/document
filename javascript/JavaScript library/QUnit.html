QUnit 

QUnit 是一個 開源(MIT) 的 javascript 單元測試框架

官網http://qunitjs.com/

使用 

1	在html中 引入 QUnit 的 js(qunit-xxx.js) css(qunit-xxx.css) 文件
2	在html中定義 分別一個id為 qunit 和 qunit-fixture 的div
3	使用 QUnit.test 定義 測試函數

example 

<!DOCTYPE html>
<html>
<head>
	<meta charset="utf-8">
	<meta name="viewport" content="width=device-width">
	<title>QUnit Example</title>
	<link rel="stylesheet" href="qunit/qunit-2.0.1.css">
</head>
<body>
	<div id="qunit"></div>
	<div id="qunit-fixture"></div>
</body>
</html>

<script src="qunit/qunit-2.0.1.js"></script>

<script type="text/javascript">
//定義一個 模塊
QUnit.module ("example1")
QUnit.test( "test-e1-1", function( assert ) {
	assert.ok( 1 == "1" ,"測試 true");
	assert.notOk(1 == "1","測試 false");

	assert.equal(1,1);
	assert.notEqual(1,2);
});
QUnit.test( "test-e1-2", function( assert ) {
	assert.ok( 1 == "1" );
});

//定義一個 模塊
QUnit.module ("example2")
QUnit.test( "test-e2-2", function( assert ) {
	assert.ok( 1 == "1");
});
</script>
