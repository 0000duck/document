main.js
$(document).ready(function() {
	var namespace = dark_using_namespace("dark.ui");
	var jq	=	$("#mylist");

	
	var obj = namespace.listimage.create(jq,{
		edit:true,
		edit_text:"請輸入新名詞",
		select:true,
		on_edit_change:function(n){
			alert("item " + n + " text change");
		},
		on_click_node:function(n){
			//alert("click " + n);
		},
		on_dbclick_node:function(n){
			//alert("dbclick " + n);
		},
		on_mouse_down:function(n,e){
			//alert("mouse_down " + n);
		},
		on_mouse_up:function(n,e){
			//alert("mouse_up " + n);
		}
	});

	obj.insert(1,{text:"3",url:"3.jpg",data:"data.3"});
	obj.append({text:"end",url:"3.jpg",data:"data.3"});

	obj.select_node(1);
	obj.select_node(3);

	//遍歷 選中 元素
	var find = obj.find_first_select();
	if(find != undefined){
		var str = "";
		do{
			str += find.n;
			str += "\n";
		}while(obj.find_next_select(find));
		alert(str);
	}

	
});
