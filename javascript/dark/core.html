//模擬 名字空間
var dark_namespace_object	=	{};		//記錄使用的	namespace

function dark_namespace(name){			//定義一個 namespace
	if(undefined	==	dark_namespace_object.name){
		dark_namespace_object.name	=	{};
	}
	return dark_namespace_object.name;
}

function dark_using_namespace(name){	//返回一個已經定義的 namespace
	return dark_namespace_object.name
}

function dark_check_namespace(name){	//檢查 namespace 是否存在
	return undefined	==	dark_namespace_object.name;
}
