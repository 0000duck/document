carousel

carousel 

定義
carousel/carousel.js

依賴
dark.core.js
dark.ui.js

namespace
dark.ui

描述
一個圖片輪播插件

常用接口 

//初始化 組件
[instance] namespace.carousel.create(_jquery	//被初始化的元素
	{begin:0	//首播圖片索引
	,interval:0	//自動翻頁時間間隔(毫秒) 0(不自動翻頁)
	})

//播放下一頁
instance.next()

//播放上一頁
instance.previous()

//顯示指定頁
instance.show_n(n)

//設置新的 自動播放時間間隔(毫秒) 0(不自動翻頁) 
instance.play(interval)

//關閉自動播放
instance.stop()

main.js
$(document).ready(function() {
	//使用 namespace
	var namespace = dark_using_namespace("dark.ui");
	var jq	=	$("#myloops");

	jq.width(800);
	jq.height(600);

	//初始化 組件
	var obj = namespace.carousel.create(jq,
		{
			begin:0		//首先播放圖片索引
			,interval:3000	//自動翻頁間隔(毫秒) 0(不主動翻頁)
		});

	
});
