获取get参数 

var url=location.href;
var pos=url.indexOf('?');

if(-1==pos){
	alert('无参');
}else{
	var str=url.substr(pos+1,url.length);
	if('#'==str[str.length-1]){
		str=str.substr(0,str.length-1);
	}
	var array=str.split('&');
	var show='';
	for(index in array){
		pos=array[index].indexOf('=');
		if(-1!=pos){
			show+=array[index].substr(pos+1,array[index].length)+"\n";
		}
	}
	alert(show);
}
