删除元素

数组排序 .sort() 

var a=new Array(1,3,2);
function sortFunction(a,b){
	if(a==b){
		return 0;
	}else if(a>b){
		//升序
    	return 1;
	}else{
    	return -1;
    }
}

a.sort(sortFunction);
//a.sort();对于基本型别默认升序
