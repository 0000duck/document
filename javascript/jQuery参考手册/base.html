each 遍歷元素 

//遍歷元素 對被個元素都 調用 function 
//返回 false 將終止遍歷後續元素
//index 為元素索引 從0開始計算 element為當前元素 (兩個都為可選參數)
$("input").each(function(index, element) {
	if(index > 0){
		return false;
		}
	alert(index);
	});

//操縱 第一個被 選中 元素 
alert($("input").attr("name"));

//設置所有 p 的值
$("input").val("123");


//返回元素個數
alert($("input").size());
alert($("input").length);

data 附加數據 

for(i=0;i<5;++i){
	//為元素 設置附加值
	$("#test_id").data("key" + i,"value" + i);
}
//獲取元素附加值
alert($("#test_id").data("key0"));

//移除指定 key 的附加值
$("#test_id").removeData("key0");
alert($("#test_id").data("key0"));
