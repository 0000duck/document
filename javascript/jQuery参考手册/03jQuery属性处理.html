attr css

attr 

//返回 name 属性
$(selector).attr("name")

//设置 name 属性为 test
$(selector).attr("name","test");

//为匹配selector的数组设置 name 属性值为 函数返回值
$(selector).attr(name,fun(index,value));

//一次设置多个属性
$(selector).attr({"name":"test","width":200});



//移除 name 属性
$(selector).removeAttr("name");

class 

//增加 test 到样式表
$(selector).addClass("test");

//从样式表中移除 test
$(selector).removeClass("test");

//切换样式表 test (既样式表中有test则移除test 否则添加test)
$(selector).toggleClass("test");

//查询样式表中是否有test
$(selector).hasClass("test");


其次class属性亦可用 attr 操作

html 

//返回/设置 html
$(selector).html();

$(selector).html(html);

//设置html为 函数返回值
$(selector).html(fun(index,value));	



//返回/设置 文本
$(selector).text();
$(selector).text("text 设置或返回 会连同所有后代元素的文本一起 返回或设置");

表单 

//返回value
$(selector).val();

//设置value
$(selector).val(value);

//设置value 为函数返回值
$(selector).val(fun(index,value));



//把 单选框 或 列表框 中value值为 value1,value2,...的选中
$(selector).val([value1,value2,...])	
	$("select[name=select_test]").value("option1","option3");

css 

//获取第一个匹配selector元素的 color
//对获取css而言 margin background border等简写形式不支持 对于background-color亦可写作backgroundColor
$(selector).css("color");


//把所有匹配selector的 border 设置为指定值
$(selector).css("border","4px solid white");

//设置name值为函数返回值
$(selector).css(name,fun(index,value));

//一次设置多个值
$(selector).css({name1:value1,name2:vales2,...})

尺寸 

//以px为单位 返回第一个匹配selector元素的实际宽度
$(selector).width();

//...高度
$(selector).height();

//设置所有匹配selector元素的宽度
$(selector).width(30);

//...高度
$(selector).height(30);



若要获取 设置元素内部区域(包括内边距padding) innerHeight innerWidth
若要获取 设置元素外部区域(包括内边距padding 边框border) outerHeight outerWidth

位置 

//返回第一个匹配selector的可见元素
var a=$(selector).offset();

//返回相对当前视窗垂直偏移
a.top;

//返回相对当前视窗水平偏移
a.left;

//设置垂直 水平 偏移
$(selector).offset({"top":40,"left":40});



//返回第一个匹配selector的可见元素
$(selector).position();	

//返回相对最近祖先元素垂直偏移
a.top;

//返回相对最近祖先元素水平偏移
a.left;



//返回第一个匹配selector元素的垂直滚动条 相对该元素顶端的偏移(没有滚动条返回0)
$(selector).scrollTop();

//...水平滚动条
$(selector).scrolleft();
