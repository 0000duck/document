jstree example

jstree 

jstree 是一個 jQuery tree 插件

官網 http://www.jstree.com/docs/html/

獲取支持 

1	從官網下載最新版本 vakata-jstree-3.0.4-0-gaa240c1.zip 並解壓
2	/dist/ 中所有 文件 放入項目
3	<link rel="stylesheet" href="dist/themes/default/style.min.css" />
	<script src="dist/jstree.min.js"></script>
4	$('#jstree_demo_div').jstree();

詳見 docs/index.html

example 

<link rel="stylesheet" href="/public/js/jstree/themes/default/style.min.css" />
<script src="/public/js/jstree/jstree.min.js"></script>

<-- tree 容器 -->
<div id="idTree">
</div>

<script type="text/javascript">
$(document).ready(function() {
	var jq = $('#idTree');
	//初始化 jsTree
	jq.jstree({
		plugins : [ 
			"wholerow",	//整行
			"checkbox",	//添加 checkbox
		],
		checkbox : {
			//three_state : false,	//對子項的選擇 不影響 祖先
		},
		core:{
			//設置 初始 節點
			data:[
				{
					id:"1",
					parent:"#",
					text:"root",
					 state:{
						opened:true,
						disabled:false,
						selected:true,
					},
				},
				{
					id:"10",
					parent:"1",
					//設置 圖標 
					icon:"/public/js/jstree/themes/default/throbber.gif",
					text:"home",
				},
				{
					id:"11",
					parent:"1",
					text:"bin",
				},
				{
					id:"101",
					parent:"10",
					text:"king",
				},
			]
		},
	});

	//返回 tree 實例 以便調用其方法
	var jqTree = jq.jstree(true);
});
</script>
