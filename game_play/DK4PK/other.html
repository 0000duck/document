威壓 

使其他勢力成為 下屬勢力

條件 

獲取 此地 霸者之證
副官說服力 [200,+∞)

此海域勢力 >= 對方總勢力 * [ 8.6 - (副官說服力 - 200) * 0.008 - 友好度 * 0.04]



lua
local dist = 700
local adjutant = 300
local friendly = 50

local std = dist * 8.6
local friendly_offset = dist * friendly * 0.04
local adjutant_offset = dist * (adjutant - 200) * 0.008
local min = std - adjutant_offset - friendly_offset


print("臨界值為	：	" , min)
print("目標勢力	：	" , dist)
print("標準值		：	" , std)
print("副官影響	：	" , dist * 0.008,"*",adjutant - 200,"=",adjutant_offset)
print("友好影響	：	" , dist * 0.04 , "*",friendly,"=",friendly_offset)
